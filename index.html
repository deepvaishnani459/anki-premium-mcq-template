Explanation
The Network Layer (Layer 3) handles logical (IP) addressing and packet routing between networks. Routers operate here. Data Link handles physical MAC addressing, Transport manages end-to-end reliability (TCP/UDP), and Session manages application sessions.

<script>
/* ═══════════════════════════════════════════════════════════════════════
   PREMIUM MCQ FRONT — Award Edition v3
   Logic: CorrectOption = EXACT TEXT of the correct answer option.
          Comparison is done by innerHTML string match.
   Sound: Web Audio API — fully embedded, zero files needed.
════════════════════════════════════════════════════════════════════════ */
(function() {
  'use strict';

  /* ── CONFIG ──────────────────────────────────────────────────────────── */
  var STREAK_KEY   = 'ankiMCQ_streak_v3';
  var LETTERS      = ['A','B','C','D'];
  var MOTIVATIONS  = [
    '⚡ Awesome!', '🚀 Keep Going!', '🎯 Spot On!', '🔥 On Fire!',
    '💪 Crushing It!', '✅ Well Done!', '🌟 Superb!',
    '🧠 Brilliant!', '💥 Let\'s Go!', '🏅 Outstanding!'
  ];
  var MILESTONES   = [
    {n:3,  e:'⭐', m:'Great Start!'},
    {n:5,  e:'🔥', m:'You\'re on Fire!'},
    {n:10, e:'🚀', m:'Amazing!'},
    {n:20, e:'🏆', m:'Elite Focus!'},
    {n:50, e:'👑', m:'Learning Machine!'}
  ];

  /* ── STATE ───────────────────────────────────────────────────────────── */
  var answered     = false;
  var wrongClicked = {};
  var correctText  = '';
  var optionEls    = [];

  /* ── STREAK ──────────────────────────────────────────────────────────── */
  function getStreak() {
    try { var v = localStorage.getItem(STREAK_KEY); return v ? +v : 0; }
    catch(e) { return 0; }
  }
  function setStreak(n) {
    try { localStorage.setItem(STREAK_KEY, n); } catch(e) {}
  }
  function renderStreak(n) {
    var el = document.getElementById('streakNum');
    if (el) el.textContent = n;
    var pill = document.getElementById('streakPill');
    if (pill) {
      pill.className = 'streak-pill' + (n > 0 ? ' streak-on' : '');
    }
  }

  /* ── WEB AUDIO SOUNDS (fully embedded — no files) ────────────────────── */
  function getAudioCtx() {
    try {
      return new (window.AudioContext || window.webkitAudioContext)();
    } catch(e) { return null; }
  }
  function playCorrectSound() {
    var ctx = getAudioCtx(); if (!ctx) return;
    var t = ctx.currentTime;
    /* Rising two-tone chime: C5 → E5 */
    [[523.25, t,      0.18, 0.0, 0.7],
     [659.25, t+0.13, 0.22, 0.0, 0.7],
     [783.99, t+0.27, 0.28, 0.0, 0.7]].forEach(function(p) {
      var osc  = ctx.createOscillator();
      var gain = ctx.createGain();
      osc.type = 'sine';
      osc.frequency.setValueAtTime(p[0], p[1]);
      gain.gain.setValueAtTime(p[4], p[1]);
      gain.gain.exponentialRampToValueAtTime(0.0001, p[1]+p[2]);
      osc.connect(gain); gain.connect(ctx.destination);
      osc.start(p[1]); osc.stop(p[1]+p[2]+0.05);
    });
  }
  function playWrongSound() {
    var ctx = getAudioCtx(); if (!ctx) return;
    var t = ctx.currentTime;
    /* Descending buzz: two low tones */
    [[220, t,      0.15, 0.0, 0.55, 'sawtooth'],
     [174, t+0.12, 0.20, 0.0, 0.45, 'sawtooth']].forEach(function(p) {
      var osc  = ctx.createOscillator();
      var gain = ctx.createGain();
      osc.type = p[5];
      osc.frequency.setValueAtTime(p[0], p[1]);
      gain.gain.setValueAtTime(p[4], p[1]);
      gain.gain.exponentialRampToValueAtTime(0.0001, p[1]+p[2]);
      osc.connect(gain); gain.connect(ctx.destination);
      osc.start(p[1]); osc.stop(p[1]+p[2]+0.05);
    });
  }

  /* ── MOTIVATIONAL POPUP ───────────────────────────────────────────────── */
  function showMotPopup(msg, cls) {
    var el = document.getElementById('motPopup');
    if (!el) return;
    el.textContent = msg;
    el.className = 'mot-popup ' + (cls || 'mot-ok') + ' mot-in';
    el.style.display = 'block';
    var t1 = setTimeout(function() {
      el.className = 'mot-popup mot-out';
      var t2 = setTimeout(function() { el.style.display = 'none'; }, 380);
    }, 1000);
  }

  /* ── MILESTONE POPUP ──────────────────────────────────────────────────── */
  function checkMilestone(n) {
    var hit = null;
    for (var i=0;i<MILESTONES.length;i++) {
      if (MILESTONES[i].n === n) { hit = MILESTONES[i]; break; }
    }
    if (!hit) return;
    var el = document.getElementById('msPopup');
    if (!el) return;
    el.innerHTML = '<span class="ms-em">'+hit.e+'</span><span class="ms-msg">'+hit.m+'</span>';
    el.className = 'ms-popup ms-in';
    el.style.display = 'flex';
    setTimeout(function() {
      el.className = 'ms-popup ms-out';
      setTimeout(function() { el.style.display='none'; },450);
    }, 1000);
  }

  /* ── CORE CLICK HANDLER ───────────────────────────────────────────────── */
  function handleClick(idx) {
    if (answered) return;
    if (wrongClicked[idx]) return;   // already wrong — ignore

    var el = optionEls[idx];
    if (!el) return;

    /* ── CORRECT ANSWER DETECTION ───────────────────────────────────────
       The CorrectOption field holds the EXACT TEXT of the correct answer.
       We compare the stored innerHTML of each option against correctText.
    ───────────────────────────────────────────────────────────────────── */
    var optHTML = el.querySelector('.opt-text-inner').innerHTML;
    var isCorrect = (optHTML === correctText);

    if (isCorrect) {
      // ✅ CORRECT
      answered = true;
      el.className = 'mcq-opt opt-correct';
      el.querySelector('.opt-icon').textContent = '✓';
      el.setAttribute('aria-pressed', 'true');

      /* lock all remaining options */
      optionEls.forEach(function(o,i) {
        if (i !== idx) { o.classList.add('opt-disabled'); o.setAttribute('aria-disabled','true'); }
      });

      /* streak */
      var streak = getStreak() + 1;
      setStreak(streak);
      renderStreak(streak);

      /* streak pulse */
      var pill = document.getElementById('streakPill');
      if (pill) {
        pill.classList.add('streak-pulse');
        setTimeout(function() { pill.classList.remove('streak-pulse'); },600);
      }

      /* motivational popup */
      showMotPopup(MOTIVATIONS[Math.floor(Math.random()*MOTIVATIONS.length)], 'mot-ok');
      checkMilestone(streak);
      playCorrectSound();

      /* hint */
      var hint = document.getElementById('mcqHint');
      if (hint) { hint.textContent = '✅ Correct! Revealing answer…'; hint.className='mcq-hint hint-ok'; }

      /* flip card — try all Anki bridge methods */
      setTimeout(function() {
        if (typeof pycmd !== 'undefined') {
          pycmd('ans');
        } else if (typeof AnkiDroidJS !== 'undefined') {
          AnkiDroidJS.markCardAsAnswered();
        } else if (typeof webkit !== 'undefined' &&
                   webkit.messageHandlers && webkit.messageHandlers.cb) {
          webkit.messageHandlers.cb.postMessage('ans');
        } else {
          /* Fallback: look for Anki's show-answer button */
          var btn = document.getElementById('ansbut') ||
                    document.querySelector('.answer-button') ||
                    document.querySelector('[id*="ans"]');
          if (btn) btn.click();
        }
      }, 750);

    } else {
      // ❌ WRONG
      wrongClicked[idx] = true;
      el.className = 'mcq-opt opt-wrong';
      el.querySelector('.opt-icon').textContent = '✗';
      el.setAttribute('aria-pressed', 'true');
      el.setAttribute('aria-disabled', 'true');
      el.classList.add('opt-disabled');

      /* reset streak */
      setStreak(0); renderStreak(0);

      playWrongSound();
      showMotPopup('❌ Try Again!', 'mot-bad');

      var hint = document.getElementById('mcqHint');
      if (hint) { hint.textContent = '❌ Incorrect — keep trying!'; hint.className='mcq-hint hint-bad'; }
    }
  }

  /* ── RENDER OPTIONS ───────────────────────────────────────────────────── */
  function renderOptions(options) {
    var container = document.getElementById('mcqOptions');
    if (!container) return;
    container.innerHTML = '';
    optionEls = [];
    options.forEach(function(opt, i) {
      var btn = document.createElement('div');
      btn.className = 'mcq-opt';
      btn.setAttribute('role', 'button');
      btn.setAttribute('tabindex', '0');
      btn.setAttribute('aria-label', 'Option ' + LETTERS[i]);
      btn.setAttribute('aria-pressed', 'false');
      btn.innerHTML =
        '<span class="opt-letter">' + LETTERS[i] + '</span>' +
        '<span class="opt-text"><span class="opt-text-inner">' + opt + '</span></span>' +
        '<span class="opt-icon"></span>';
      (function(idx){ btn.addEventListener('click', function(){ handleClick(idx); }); })(i);
      btn.addEventListener('keydown', (function(idx){ return function(e){
        if (e.key==='Enter'||e.key===' ') { e.preventDefault(); handleClick(idx); }
      }; })(i));
      container.appendChild(btn);
      optionEls.push(btn);
    });
  }

  /* ── INIT ─────────────────────────────────────────────────────────────── */
  function init() {
    var dQ = document.getElementById('dQ'),
        d1 = document.getElementById('d1'),
        d2 = document.getElementById('d2'),
        d3 = document.getElementById('d3'),
        d4 = document.getElementById('d4'),
        dC = document.getElementById('dC');

    if (!dQ||!d1||!dC) return;

    /* populate question */
    var qEl = document.getElementById('mcqQuestion');
    if (qEl) qEl.innerHTML = dQ.innerHTML.trim();

    /* ── THE CORRECT ANSWER KEY ─────────────────────────────────────────
       correctText = the innerHTML of the CorrectOption field.
       This MUST match one of the option HTML strings exactly.
       The user puts the exact option text into CorrectOption.
    ──────────────────────────────────────────────────────────────── */
    correctText = dC.innerHTML.trim();

    /* collect non-empty options */
    var raw = [d1,d2,d3,d4].map(function(el){ return el.innerHTML.trim(); })
                            .filter(function(t){ return t !== ''; });

    if (raw.length === 0) return;
    if (!correctText) {
      console.warn('MCQ: CorrectOption field is empty!'); return;
    }

    /* shuffle */
    var options = raw.slice();
    for (var i=options.length-1; i>0; i--) {
      var j = Math.floor(Math.random()*(i+1));
      var tmp = options[i]; options[i]=options[j]; options[j]=tmp;
    }

    renderOptions(options);
    renderStreak(getStreak());
  }

  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', function() {
      requestAnimationFrame(init);
    });
  } else {
    requestAnimationFrame(init);
  }

})();
</script>

/* ═══════════════════════════════════════════════════════════════════════
   BACK SIDE INIT
   Reads CorrectOption text, highlights matching option in list.
════════════════════════════════════════════════════════════════════════ */
(function() {
  'use strict';
  var STREAK_KEY = 'ankiMCQ_streak_v3';
  var LETTERS    = ['A','B','C','D'];

  function getStreak() {
    try { var v=localStorage.getItem(STREAK_KEY); return v ? +v : 0; }
    catch(e){ return 0; }
  }

  function run() {
    var bdQ=document.getElementById('bdQ'),
        bd1=document.getElementById('bd1'),
        bd2=document.getElementById('bd2'),
        bd3=document.getElementById('bd3'),
        bd4=document.getElementById('bd4'),
        bdC=document.getElementById('bdC');
    if (!bdC) return;

    /* question */
    var qEl=document.getElementById('backQuestion');
    if (qEl && bdQ) qEl.innerHTML=bdQ.innerHTML.trim();

    /* correct text — must match exactly */
    var correctText = bdC.innerHTML.trim();

    /* options (keep original order on back) */
    var opts = [bd1,bd2,bd3,bd4]
      .map(function(el){ return el ? el.innerHTML.trim() : ''; })
      .filter(function(t){ return t !== ''; });

    var container = document.getElementById('backOptions');
    if (!container) return;
    container.innerHTML = '';

    opts.forEach(function(text, i) {
      var isCorrect = (text === correctText);
      var row = document.createElement('div');
      row.className = 'back-opt-row' + (isCorrect ? ' back-opt-correct' : '');
      row.setAttribute('role', 'listitem');
      row.innerHTML =
        '<span class="back-ltr">' + LETTERS[i] + '</span>' +
        '<span class="back-opt-text">' + text + '</span>' +
        (isCorrect ? '<span class="back-opt-tick">✓</span>' : '');
      container.appendChild(row);
    });

    /* streak */
    var streak = getStreak();
    var numEl   = document.getElementById('backStreakNum');
    if (numEl) numEl.textContent = streak;
    var badge   = document.getElementById('backStreakBadge');
    if (badge && streak > 0) badge.classList.add('badge-live');
  }

  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', function() { requestAnimationFrame(run); });
  } else {
    requestAnimationFrame(run);
  }
})();
</script>

Setup Guide
How to Install in Anki
1
In Anki: Tools → Manage Note Types → Add a new note type called MCQ Premium.
2
Add exactly 7 fields (case-sensitive): Question · Option1 · Option2 · Option3 · Option4 · CorrectOption · Explanation
3
Click Cards… → paste Front Template, Back Template, and Styling into their respective sections.
4
⭐ KEY — CorrectOption field: Enter the exact text of the correct answer, copied from one of the Option fields. Example:
Option2 = Network Layer → CorrectOption = Network Layer
This is the same proven logic as your original template.
5
Sounds: Fully embedded via Web Audio API — no mp3 files needed. Works everywhere instantly.
6
Save and start reviewing. Streak persists across sessions via localStorage.
⚠️ CorrectOption must exactly match the option text. Copy-paste from Option1/2/3/4 to ensure no hidden spaces or formatting differences.
✅ Sounds work via Web Audio API — generates tones programmatically. No files, no CDN, no internet. A rising 3-note chime for correct, a descending buzz for wrong. Gracefully silent if Web Audio is unavailable.
